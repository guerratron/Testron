Ejemplos Testron: con diferentes objetos de entrada

Testron a través de distintos objetos con estructura JSON

Seleccionar una forma para parsear una "batería de preguntas":
  Constructor.     Objeto.     Cadena. 
//CONSTRUCTOR
var test = new testron.Test(
  {container: "divUI", markerContainer: "divMarker", 
  stopOnSuccess: false, digits: 2, precision: 2}
);
if (testron) {
  var question1 = test.createQuestion({num: 1, solution: 2, txt: "Pregunta 1 ???"});
  var answer1A = question1.createAnswer({name: "A", txt: "Respuesta A"});
  var answer1B = question1.createAnswer({name: "B", txt: "Respuesta B"});
  var answer1C = question1.createAnswer({name: "C", txt: "Respuesta C"});
  
  var question2 = test.createQuestion({num: 2, solution: 3, txt: "Pregunta 2 ???"});
  var answer2A = question2.createAnswer({name: "A", txt: "Respuesta A"});
  var answer2B = question2.createAnswer({name: "B", txt: "Respuesta B"});
  var answer2C = question2.createAnswer({name: "C", txt: "Respuesta C"});
  
  var question3 = test.createQuestion({num: 3, solution: 3, txt: "Pregunta 3 ???"});
  var answer3A = question3.createAnswer({name: "A", txt: "Respuesta A"});
  var answer3B = question3.createAnswer({name: "B", txt: "Respuesta B"});
  var answer3C = question3.createAnswer({name: "C", txt: "Respuesta C"});
  
  var question4 = test.createQuestion({num: 4, solution: 1, txt: "Pregunta 4 ???"});
  var answer4A = question4.createAnswer({name: "A", txt: "Respuesta A"});
  var answer4B = question4.createAnswer({name: "B", txt: "Respuesta B"});
  var answer4C = question4.createAnswer({name: "C", txt: "Respuesta C"});

  var container = document.getElementById("contUI");
  container.appendChild(test.toHtml());
  var marker = test.createMarker();
  if (marker) { container2.appendChild(marker.toHtml()); }
  //START
  test.emitMarker();
}

Importar en la sección Header-HTML un archivo de test con una batería de preguntas, eg:
  <script src="js/test3-star-wars.js"> </script>
y escribir el script:
  tests = starWars; //aquí "starWars" es la variable en el archivo importado.
  //TESTRON
  tests = testron.parseTestrons(tests, conf);
